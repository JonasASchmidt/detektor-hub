Create `app/dashboard/settings/page.tsx` as a server component that fetches the user's current default and renders a client sub-component for the interactive toggle.

The file should contain both the server component (default export) and the client sub-component in the same file (or split into page.tsx + SettingsClient.tsx — use a single file for simplicity since it's small).

Pattern to follow:
```typescript
// app/dashboard/settings/page.tsx
import { getServerSession } from "next-auth";
import { authOptions } from "@/lib/auth";
import prisma from "@/lib/prisma";
import { redirect } from "next/navigation";
import SettingsClient from "./SettingsClient";  // OR inline "use client" component

export default async function SettingsPage() {
  const session = await getServerSession(authOptions);
  if (!session?.user?.id) redirect("/");

  const user = await prisma.user.findUnique({
    where: { id: session.user.id },
    select: { defaultShowFinderName: true },
  });

  return (
    <div className="p-6 md:pt-12 space-y-6 max-w-3xl mx-auto">
      <h1 className="text-4xl font-bold">Einstellungen</h1>
      <SettingsClient defaultShowFinderName={user?.defaultShowFinderName ?? false} />
    </div>
  );
}
```

Create the client component (either inline in the same file with a named export, or in a separate `SettingsClient.tsx` in the same directory):
```typescript
"use client";
import { useState } from "react";
import { Switch } from "@/components/ui/switch";
import { Label } from "@/components/ui/label";
import { Card } from "@/components/ui/card";

export default function SettingsClient({
  defaultShowFinderName,
}: {
  defaultShowFinderName: boolean;
}) {
  const [value, setValue] = useState(defaultShowFinderName);

  const handleToggle = async (checked: boolean) => {
    setValue(checked);
    await fetch("/api/user", {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ defaultShowFinderName: checked }),
    });
  };

  return (
    <Card className="bg-white dark:bg-gray-900">
      <div className="py-4 px-6">
        <div className="flex items-center gap-3">
          <Switch
            checked={value}
            onCheckedChange={handleToggle}
            id="defaultShowFinderName"
          />
          <Label htmlFor="defaultShowFinderName">
            Standard: Namen bei neuen Funden öffentlich anzeigen
          </Label>
        </div>
      </div>
    </Card>
  );
}
```

If placing SettingsClient in a separate file, create `app/dashboard/settings/SettingsClient.tsx`. If inlining as a named export in page.tsx, use "use client" directive only on the inner component (not the page file) — the cleanest approach is a separate SettingsClient.tsx file.
