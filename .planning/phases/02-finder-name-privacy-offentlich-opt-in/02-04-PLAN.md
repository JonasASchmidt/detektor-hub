import { getServerSession } from "next-auth";
import { authOptions } from "@/lib/auth";
import prisma from "@/lib/prisma";
import { redirect, notFound } from "next/navigation";
import EditFindingForm from "./EditFindingForm";

interface Props {
  params: Promise<{ id: string }>;
}

export default async function EditFindingPage({ params }: Props) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.id) redirect("/");

  const { id } = await params;

  const finding = await prisma.finding.findUnique({
    where: { id },
    select: {
      id: true,
      name: true,
      showFinderName: true,
      userId: true,
    },
  });

  if (!finding) notFound();

  // Ownership check — redirect non-owners
  if (finding.userId !== session.user.id) redirect(`/dashboard/findings/${id}`);

  return (
    <div className="p-6 md:pt-12 space-y-6 max-w-3xl mx-auto">
      <h1 className="text-4xl font-bold">Fund bearbeiten</h1>
      <EditFindingForm
        findingId={finding.id}
        findingName={finding.name ?? ""}
        initialShowFinderName={finding.showFinderName}
      />
    </div>
  );
}

```typescript
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
import { Switch } from "@/components/ui/switch";
import { Label } from "@/components/ui/label";
import { Button } from "@/components/ui/button";
import { Card } from "@/components/ui/card";
import { Loader2 } from "lucide-react";
import { toast } from "sonner";

interface Props {
  findingId: string;
  findingName: string;
  initialShowFinderName: boolean;
}

export default function EditFindingForm({
  findingId,
  findingName,
  initialShowFinderName,
}: Props) {
  const router = useRouter();
  const [showFinderName, setShowFinderName] = useState(initialShowFinderName);
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setLoading(true);

    const res = await fetch(`/api/findings/${findingId}`, {
      method: "PUT",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ showFinderName }),
    });

    setLoading(false);

    if (!res.ok) {
      toast.error("Änderungen konnten nicht gespeichert werden.");
      return;
    }

    toast.success("Änderungen gespeichert.");
    router.push(`/dashboard/findings/${findingId}`);
  };

  return (
    <form onSubmit={handleSubmit} className="flex flex-col gap-4">
      <Card className="bg-white dark:bg-gray-900">
        <div className="py-4 px-6 space-y-4">
          <p className="font-semibold text-sm text-muted-foreground">
            {findingName}
          </p>
          <div className="flex items-center gap-3">
            <Switch
              checked={showFinderName}
              onCheckedChange={setShowFinderName}
              id="showFinderName"
            />
            <Label htmlFor="showFinderName">
              Meinen Namen bei diesem Fund öffentlich anzeigen
            </Label>
          </div>
        </div>
      </Card>
      {loading ? (
        <Button disabled>
          <Loader2 className="animate-spin" />
          Bitte Warten
        </Button>
      ) : (
        <Button type="submit">Speichern</Button>
      )}
    </form>
  );
}
```

Note: This form uses local `useState` for the Switch instead of react-hook-form Controller, since it's a minimal form with only one boolean field. No Zod schema needed client-side here — validation is on the API.
